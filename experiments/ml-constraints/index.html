// Python - PyTorch Integration
from constraint_theory import ConstraintEnforcedLayer, GradientSnapper
import torch

# Create constraint-enforced layer
layer = ConstraintEnforcedLayer(
    in_features=768,
    out_features=256,
    constraint='unit_norm',  # Options: unit_norm, orthogonal, bounded
    epsilon=1e-6
)

# Training with gradient snapping
snapper = GradientSnapper(preserve_magnitude=True)

for batch in dataloader:
    loss = model(batch)
    loss.backward()
    
    # Snap gradients to exact before optimizer step
    snapper.snap_(model.parameters())
    
    optimizer.step()
